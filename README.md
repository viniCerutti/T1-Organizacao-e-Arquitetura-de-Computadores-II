# Ordem do endereçamento do Driver na memória

0xffff0000 - Tx_Data <br />
0xffff0004 - tx_av <br />
0xffff0008 - rx_data <br />
0xffff000C - rx_start <br />
0xffff0010 - rx_busy <br />

# O que precisa Fazer
